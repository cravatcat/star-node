# 10.1 useRequest / useFetch

## 1. 实现思路

封装 fetch 请求，处理 loading、error、data 状态，支持重试和取消。

## 2. 代码实现

```typescript
import { useState, useEffect, useCallback } from 'react';

interface UseRequestOptions {
  manual?: boolean;
  onSuccess?: (data: any) => void;
  onError?: (error: any) => void;
}

function useRequest(url: string, options: UseRequestOptions = {}) {
  const [data, setData] = useState<any>(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<any>(null);

  const run = useCallback(async () => {
    setLoading(true);
    setError(null);
    try {
      const response = await fetch(url);
      if (!response.ok) throw new Error('Network response was not ok');
      const result = await response.json();
      setData(result);
      options.onSuccess?.(result);
    } catch (err) {
      setError(err);
      options.onError?.(err);
    } finally {
      setLoading(false);
    }
  }, [url]);

  useEffect(() => {
    if (!options.manual) {
      run();
    }
  }, [run, options.manual]);

  return { data, loading, error, run };
}

export default useRequest;
```

## 3. 使用示例

```tsx
import useRequest from './useRequest';

function App() {
  const { data, loading, error, run } = useRequest('https://api.example.com/data');

  if (loading) return <div>Loading...</div>;
  if (error) return <div>Error: {error.message} <button onClick={run}>Retry</button></div>;
  
  return <div>Data: {JSON.stringify(data)}</div>;
}
```

## 4. 常见问题

- **如何取消请求？** 可以使用 AbortController。
- **如何处理依赖刷新？** 将依赖项作为 useEffect 的依赖。
